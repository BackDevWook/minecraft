# Bukkit `Material` Enum 목록

public enum Material {
    // 기본 블록
    AIR, STONE, GRASS, DIRT, COBBLESTONE, WOOD, SAPLING,
    BEDROCK, WATER, LAVA, SAND, GRAVEL, GOLD_ORE, IRON_ORE, COAL_ORE,
    // 나무 관련
    ACACIA_LOG, BIRCH_LOG, JUNGLE_LOG, SPRUCE_LOG,
    // 금속 블록
    GOLD_BLOCK, IRON_BLOCK, DIAMOND_BLOCK, EMERALD_BLOCK,

    ENCHANTING_TABLE, BEACON, OBSIDIAN, REDSTONE_LAMP_OFF,
    // 도구 및 무기
    WOODEN_SWORD, STONE_SWORD, IRON_SWORD, GOLDEN_SWORD, DIAMOND_SWORD,
    // 곡괭이
    WOODEN_PICKAXE, STONE_PICKAXE, IRON_PICKAXE, GOLDEN_PICKAXE, DIAMOND_PICKAXE,
    // 삽
    WOODEN_SHOVEL, STONE_SHOVEL, IRON_SHOVEL, GOLDEN_SHOVEL, DIAMOND_SHOVEL,
    // 도끼
    WOODEN_AXE, STONE_AXE, IRON_AXE, GOLDEN_AXE, DIAMOND_AXE,
    // 곡괭이
    WOODEN_HOE, STONE_HOE, IRON_HOE, GOLDEN_HOE, DIAMOND_HOE,
    // 방어구
    LEATHER_HELMET, CHAINMAIL_HELMET, IRON_HELMET, GOLDEN_HELMET, DIAMOND_HELMET,
    LEATHER_CHESTPLATE, CHAINMAIL_CHESTPLATE, IRON_CHESTPLATE, GOLDEN_CHESTPLATE, DIAMOND_CHESTPLATE,
    LEATHER_LEGGINGS, CHAINMAIL_LEGGINGS, IRON_LEGGINGS, GOLDEN_LEGGINGS, DIAMOND_LEGGINGS,
    LEATHER_BOOTS, CHAINMAIL_BOOTS, IRON_BOOTS, GOLDEN_BOOTS, DIAMOND_BOOTS,
    // 음식 및 소비 아이템
    APPLE, BREAD, GOLDEN_APPLE, COOKED_BEEF, COOKED_CHICKEN, PORKCHOP,
    COOKED_PORKCHOP, COOKED_FISH, MUSHROOM_STEW, COOKIE, CAKE,
    // 기타 아이템
    BUCKET, MILK_BUCKET, WATER_BUCKET, LAVA_BUCKET,
    ENDER_PEARL, BOOK, WRITTEN_BOOK, COMPASS,
    FISHING_ROD, BOW, ARROW,
    REDSTONE, TORCH, TNT, CHEST, FURNACE, WORKBENCH, ANVIL,
    // 몬스터 알, 기록 등
    ZOMBIE_SPAWN_EGG, CREEPER_SPAWN_EGG, SKELETON_SPAWN_EGG,
    MUSIC_DISC_13, MUSIC_DISC_CAT, MUSIC_DISC_BLOCKS,
    // 목재 종류의 배, 도어, 펜스
    ACACIA_BOAT, BIRCH_BOAT, JUNGLE_BOAT, SPRUCE_BOAT,
    ACACIA_DOOR, BIRCH_DOOR, JUNGLE_DOOR, SPRUCE_DOOR,
    ACACIA_FENCE, BIRCH_FENCE, JUNGLE_FENCE, SPRUCE_FENCE,
    ACACIA_FENCE_GATE, BIRCH_FENCE_GATE, JUNGLE_FENCE_GATE, SPRUCE_FENCE_GATE,
